222bool addNeighbor(GraphNode*** node_2D_array, int i, int j, int direction, int scent) { // Adds the neighbor of the [i][j] tile in the specified direction
223 int new_row, new_col;
224 int scent_change = NO_CHANGE;
225 int distance = 1 + (direction/4); // Only used for purple tiles, Allow the function to be called recursively by adding 4 to the direction for each purple tile in a row
226 direction = direction % 4;
227
228 // Determine the new coordinates based on the direction
229 switch (direction)
230 {
231 case EAST:
232 new_row = j;
233 new_col = i + distance; // Almost always i+1 unless called resursively for purple tiles
234 break;
235 case NORTH:
236 new_row = j - distance;
237 new_col = i;
238 break;
239 case WEST:
240 new_row = j;
241 new_col = i - distance;
242 break;
243 case SOUTH:
244 new_row = j + distance;
245 new_col = i;
246 break;
247 default:
248 assert(false);
249 return false; // Invalid direction, do nothing
250 }
251 if (new_row < 0 || new_row >= maze_width) return false;
252 if (new_col < 0 || new_col >= maze_height) return false;
253
254 GraphNode* current = node_2D_array[i][j];
255 GraphNode* neighbor = node_2D_array[new_col][new_row];
256
257 if (neighbor->color == RED || neighbor->color == YELLOW) return false;
258 if (neighbor->color == BLUE) // If the neighbor is blue, we check for every adjacent tile if one of them is yellow. If it is the case, then we skip this link because the tile is blocking
259 {
260 for (int scan_direction = EAST; scan_direction <= SOUTH; scan_direction++)
261 {
262 if (detectColor(node_2D_array, new_col, new_row, scan_direction) == YELLOW) return false;
263 }
264 if (scent == ORANGE) return false; // If the scent is orange the blue tile cannot be walked on, so we don't link it
265 }
266 if (neighbor->color == COLOR_ORANGE) scent_change = ORANGE;
267 if (neighbor->color == PURPLE) // If the neighbor is purple, we try to link the node to the one further up in that direction. If it fails because of a wall, then we link the node to itself (since the player will bounce)
268 {
269 scent_change = LEMON;
270 if (!addNeighbor(node_2D_array, i, j, ((distance)*4)+direction, scent)) // Try to link the node to the one further up in that direction
271 {
272 addNeighborToNode(current, current, scent, scent_change, direction);
273 }
274 return true;
275 }
276 if (distance > 1) scent_change = LEMON; // If the tiles aren't adjacent it means there is a purple tile between them
277 addNeighborToNode(current, neighbor, scent, scent_change, direction);
278 return true;
279}

392char *findShortestPath(Graph *graph, GraphNode *current_node, GraphNode *end_node, int scent)
393{
394 if (graph == NULL || current_node == NULL || end_node == NULL) return NULL; // If there is a problem we simply exit
395 if (current_node == end_node) return ""; // If we are at the end node, we return an empty string to indicate that
396
397 int neighbor_count; // We declare the number of neighbors
398 NodeConnection **neighbors; // We declare the neighbors
399
400 if (scent == ORANGE)
401 {
402 current_node->visited_orange = true; // We mark the current node as visited
403 neighbor_count = current_node->neighbor_count_orange; // We get the number of neighbors
404 neighbors = current_node->neighbors_orange; // We get the neighbors
405 }
406 else if (scent == LEMON)
407 {
408 current_node->visited_lemon = true; // We mark the current node as visited
409 neighbor_count = current_node->neighbor_count_lemon; // We get the number of neighbors
410 neighbors = current_node->neighbors_lemon; // We get the neighbors
411 }
412 char **path = (char**)malloc(neighbor_count * sizeof(char*)); // We declare the path list whose shortest member will be returned
413 char direction_text[4];// = "\0"; // We declare the direction text
414
415
416 for (int i = 0; i < neighbor_count; i++) // For each of the current node's neighbors
417 {
418 path[i] = (char*)malloc(1000 * sizeof(char)); // We allocate memory for the path
419 if (path[i] == NULL) return NULL; // If there is a problem we exit
420 for (int j = 0; j < 1000; j++) path[i][j] = 'R'; // !! TEMPORARY SOLUTION, so that a path which got skipped by the continue statement doesn't get chosen by findShortestString
421 NodeConnection *neighbor = neighbors[i]; // We declare neighbor, the neighbor that is being checked
422 if (neighbor->scent_change != NO_CHANGE) scent = neighbor->scent_change; // We change the scent if needed (we go to an orange or purple tile)
423 if (scent == ORANGE && neighbor->node->visited_orange) continue; // If already visited with the current_scent, we ignore this node
424 if (scent == LEMON && neighbor->node->visited_lemon) continue;
425 path[i][0] = '\0'; // If the path didn't get skipped, we initialise it normally
426 sprintf(path[i] + strlen(path[i]), "%s", findShortestPath(graph, neighbor->node, end_node, scent)); // We call the function recursively to ensure that every node will be checked
427 sprintf(direction_text, "%d", neighbor->cardinal_direction); // We get the direction of the neighbor
428 strcat(path[i], direction_text); // We concatenate the direction with the path
429 }
430 if (path == NULL) return NULL; // If we have not found a path, we return NULL
431 return path[findShortestString(path, neighbor_count)]; // We return the shortest path
432}
